simplelikes

Drop-in anonymous likes for your static site.

Custom text examples

Global config (window.__simpleLikesConfig) is set below — tags without inline attributes use "star"/"stars". Inline attributes win over the global config.

Usage

<!-- Inline (takes precedence) -->
<simple-likes slug="pt-post" text="coração" text-plural="corações"></simple-likes>
<simple-likes slug="clap-post" text="clap"></simple-likes>

<!-- Global config (fallback for tags without inline attributes) -->
<script>
  window.__simpleLikesConfig = {
    apiUrl: "https://likes.yourdomain.com",
    text: "star",
    "text-plural": "stars",
  };
</script>

<script src="dist/simple-likes.js"></script>
